Function GetNamedRangeRefersTo(namedRangeName As String) As String
    ' Declare a variable to store the RefersTo value
    Dim refersTo As String

    ' Get the RefersTo property of the named range
    refersTo = ThisWorkbook.Names(namedRangeName).RefersTo

    ' Return the refersTo value
    GetNamedRangeRefersTo = refersTo
End Function
